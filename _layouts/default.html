(function () { var canvas = document.getElementById('logo'); var ctx = canvas.getContext('2d'); var W = canvas.width; var H = canvas.height; // Outer equilateral triangle vertices var pad = 6; var T = { x: W / 2, y: pad }; var BL = { x: pad, y: H - pad }; var BR = { x: W - pad, y: H - pad }; // Midpoints (medial triangle) var mTBL = { x: (T.x + BL.x) / 2, y: (T.y + BL.y) / 2 }; var mTBR = { x: (T.x + BR.x) / 2, y: (T.y + BR.y) / 2 }; var mBLR = { x: (BL.x + BR.x) / 2, y: (BL.y + BR.y) / 2 }; // Colors — three teal-green shades, none identical to HTB palette var C1 = '#006e66'; // top corner — deep teal var C2 = '#009978'; // bottom-left — medium teal-green var C3 = '#004d52'; // bottom-right — dark ocean var CC = '#0d1117'; // centre — background function tri(p1, p2, p3, fill) { ctx.beginPath(); ctx.moveTo(p1.x, p1.y); ctx.lineTo(p2.x, p2.y); ctx.lineTo(p3.x, p3.y); ctx.closePath(); ctx.fillStyle = fill; ctx.fill(); } // Four medial sections tri(T, mTBL, mTBR, C1); // top corner tri(BL, mBLR, mTBL, C2); // bottom-left corner tri(BR, mTBR, mBLR, C3); // bottom-right corner tri(mTBL, mTBR, mBLR, CC); // centre (inverted) // Thin accent borders on all inner lines + outer edge ctx.strokeStyle = '#0d1117'; ctx.lineWidth = 3; ctx.lineJoin = 'round'; ctx.beginPath(); ctx.moveTo(T.x, T.y); ctx.lineTo(BL.x, BL.y); ctx.lineTo(BR.x, BR.y); ctx.closePath(); ctx.stroke(); ctx.strokeStyle = '#0d1117'; ctx.lineWidth = 2; [[mTBL, mTBR], [mTBL, mBLR], [mTBR, mBLR]].forEach(function(pair) { ctx.beginPath(); ctx.moveTo(pair[0].x, pair[0].y); ctx.lineTo(pair[1].x, pair[1].y); ctx.stroke(); }); // Outer accent glow ctx.strokeStyle = '#00e5a0'; ctx.lineWidth = 1.5; ctx.beginPath(); ctx.moveTo(T.x, T.y); ctx.lineTo(BL.x, BL.y); ctx.lineTo(BR.x, BR.y); ctx.closePath(); ctx.stroke(); // Centre ∆ symbol var cx = (mTBL.x + mTBR.x + mBLR.x) / 3; var cy = (mTBL.y + mTBR.y + mBLR.y) / 3; ctx.fillStyle = '#00e5a0'; ctx.font = 'bold 18px JetBrains Mono, monospace';
